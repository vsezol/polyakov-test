// js
window.onload = () => {
 addHiddenIframe();
}

function addHiddenIframe() {
 const iframe = document.createElement('iframe')
 iframe.src = 'https://vsezol.com/polyakov-test';
 iframe.width = 600;
 iframe.height = 200;
 iframe.classList.add('hidden-iframe');
 
 document.body.append(iframe);
}

// css
.hidden-iframe {
    position: fixed;
    bottom: 5px;
    left: 10px;
    z-index: 20000;
    opacity: 0.0;
}

.hidden-iframe:hover {
    opacity: 1.0;

}
